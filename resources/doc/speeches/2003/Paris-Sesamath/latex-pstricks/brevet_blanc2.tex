\begin{enumerate}
        \item Observer la mise en équation de ce problème~:
        \begin{center}
                \fbox{\begin{minipage}{8cm}
                Soit \emph{a} le nombre de billes d'Antoine, et \emph{t} le nombre de billes
                de Thomas~:
                $$\left\lbrace
                        \begin{array}{rcl}
                        a+6 & = & t-6 \\
                        a+10 & = & 2(t-10)
                        \end{array}\right.$$
                \end{minipage}}
        \end{center}
        Recopier l'énoncé du problème en le complétant par les nombres qui manquent.\point{1}
        \item Calculer le nombre de billes d'Antoine et de Thomas.\points{2}
\end{enumerate}
